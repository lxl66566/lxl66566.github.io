function sanitizeSlotName(str: string) {
  // 正则表达式匹配所有非中文、日文、数字、字母的字符
  const regex = /[^a-zA-Z0-9\u4e00-\u9fa5\u3040-\u30ff]/g;
  return str.replace(regex, "");
}
